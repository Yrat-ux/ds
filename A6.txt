#include <bits/stdc++.h>
using namespace std;

int n;
vector<vector<int>> board;

// Check if we can place a queen at (row, col)
bool isSafe(int row, int col) {
    // Check column
    for (int i = 0; i < row; i++)
        if (board[i][col] == 1)
            return false;

    // Check upper-left diagonal
    for (int i = row, j = col; i >= 0 && j >= 0; i--, j--)
        if (board[i][j] == 1)
            return false;

    // Check upper-right diagonal
    for (int i = row, j = col; i >= 0 && j < n; i--, j++)
        if (board[i][j] == 1)
            return false;

    return true;
}

// Recursively place queens row by row
bool solve(int row) {
    if (row == n)   // All queens are placed
        return true;

    for (int col = 0; col < n; col++) {
        if (isSafe(row, col)) {
            board[row][col] = 1;   // Place queen

            if (solve(row + 1))    // Recur
                return true;

            board[row][col] = 0;   // Backtrack
        }
    }
    return false;
}

// Print board
void printBoard() {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            cout << (board[i][j] ? " Q " : " . ");
        cout << endl;
    }
}

int main() {
    cout << "Enter value of N: ";
    cin >> n;

    board.assign(n, vector<int>(n, 0));

    if (solve(0))
        printBoard();
    else
        cout << "No solution exists for N = " << n;

    return 0;
}
